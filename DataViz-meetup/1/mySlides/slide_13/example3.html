<!-- Example 3: My Meetup groups with d3.js -->
<!DOCTYPE html>
  <meta charset="utf-8"> <!-- also save this file as unicode-8 ! -->
  <head>
    <script src="http://d3js.org/d3.v3.js"></script>
  </head>
  
  <body> 
    <h1>My meetup groups are:</h1>
    
    <script>
      var meetupGroups 
	  = ["Big Data", "Big Data Developers", "Streaming Big Data"];      
      display(meetupGroups); 
	  
      function display(mydata){
        var anchor = d3.select("body");
        
        var selection = anchor.selectAll("p")
          .data(mydata);
		  
        // update selection  
        selection.text(function (d) { return d + " - updated";}); 
        
        selection.enter() // enter selection
          .append("p")
          .text(function (d) { return d + " - entered";});
          
        selection.exit() // exit selection
          .remove();
      }     
    </script>
  </body>
</html>
